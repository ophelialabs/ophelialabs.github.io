Code Example

-- Top Earners by Department and City
SELECT department, city, MAX(salary) AS top_earner
FROM employees
GROUP BY department, city
ORDER BY department, city;

Example 2:

WITH ranked AS (
    Select
    id, name, city, department, salary,
    Rank() OVER (
        PARTITION BY department
        ORDER BY salary DESC
    ) AS dept_rank,
    Rank() OVER (
        PARTITION BY city
        ORDER BY salary DESC
    ) AS city_rank
    From employees
)Select
    id, name, city, department, salary, dept_rank, city_rank,
    CASE WHEN dept_rank = 1 THEN 1 ELSE 0 END AS is_top_in_dept,
    CASE WHEN city_rank = 1 THEN 1 ELSE 0 END AS is_top_in_city
From ranked;
WHERE dept_rank = 1 OR city_rank = 1;
ORDER BY department, city, salary DESC, name;
